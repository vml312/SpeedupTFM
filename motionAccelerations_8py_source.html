56 def optical_flow_dense_from_video(video_name, frame_skip):
57  cap = cv2.VideoCapture(video_name)
58  # Get the first frame and convert it to grayscale
59  ret, frame1 = cap.read()
60  prvs = cv2.cvtColor(frame1, cv2.COLOR_BGR2GRAY)
61 
62  mag_list = []
63  frame_count = 0
64  while True:
65  ret, frame2 = cap.read()
66  if not ret:
67  break
68 
69  frame_count += 1
70  if frame_skip:
71  if frame_count % frame_skip != 0:
72  continue
73 
74  next_frame = cv2.cvtColor(frame2, cv2.COLOR_BGR2GRAY)
75  # Calculates dense optical flow by Farneback method
76  flow = cv2.calcOpticalFlowFarneback(prvs, next_frame, None, 0.5, 3, 15, 3, 5, 1.2, 0)
77  magnitude, angle = cv2.cartToPolar(flow[..., 0], flow[..., 1])
78  # Define the window size for the moving average
79  window_size = 15
80  step_size = 5
81  mean_mags = []
82  # Apply the moving average
83  for i in range(0, magnitude.shape[0] - window_size, step_size):
84  window = magnitude[i : i + window_size]
85  mean_mag = np.mean(window)
86  mean_mags.append(mean_mag)
87  # Analyze the mean magnitudes to identify high-motion regions
88  if len(mean_mags)>0:
89  high_motion_threshold = np.percentile(mean_mags, 90) # Threshold for high motion
90  high_motion_windows = [mag for mag in mean_mags if mag > high_motion_threshold]
91  mean95 = np.mean(high_motion_windows)
92  mag_list.append(mean95)
93 
94  prvs = next_frame
95  cap.release()
96  cv2.destroyAllWindows()
97 
98  return mag_list

156 def get_acc_from_limits(value, acc_max, acc_min, value_max, value_min):
157  acc = acc_max - ((acc_max - acc_min)/(value_max - value_min))*(value - value_min)
158  if np.isneginf(acc) or np.isposinf(acc) or np.isnan(acc):
159  acc = acc_max

177 def time_series_subsegments(df, min_video_duration, percentile_high, percentile_low, acc_max, acc_min, value_max, value_min):
178  state = "low"
179  error = 0

188 
189  acc_interval = 0
190

192  df["acc"] = acc_min
193  error = 1

199  error = 1

201 
202  for i in df.index:
203

205

207

209  df.loc[i, "acc-interval"] = acc_interval

221  elif state == "high":

224  state = "low"

228  df.loc[i, "acc"] = acc
229  df.loc[i, "acc-interval"] = acc_interval

250 # First, the program format_ffmpeg_scene_cut is run to obtain the changes of scene of the video and each scene is assigned its number in the ''interval'' column. The acceleration
251 # of each processed frame is then adjusted so that the duration of the accelerated scenes is greater than the parameter entered in the configuration file (''min_acc_scene_duration'')
252 # Since the correction has potentially varied the acceleration value of all the frames in them, the average of all the accelerations that are part of a scene is taken (correct_acc_min_video_duration)
253 # @param path The path where the video files are stored
254 # @param filename The name of the video file
255 # @param scene_cut_threshold The threshold to detect scene cuts
256 # @param df The dataframe with the optical flow values
257 # @param min_acc_scene_duration The minimum accelerated scene duration
258 # @return The dataframe with the corrected accelerations

260 def correct_acc_from_scene_cuts(path, filename, scene_cut_threshold, df, min_acc_scene_duration):
261 
262  scene_cut_times = format_ffmpeg_scene_cut.main(path, filename, scene_cut_threshold)
263 
264  #Time case is 0.101 and skipping 5 frames, first time is 0.2
265  if df.loc[0, "time-s"]>scene_cut_times[0]:
266  scene_cut_times[0] = df.loc[0, "time-s"]
267 
268  index_df = 0
269  interval_time=0
270 
271  for count, left_interval in enumerate(scene_cut_times):
272  while index_df < len(df) and df.loc[index_df, "time-s"]<=left_interval:
273  df.loc[index_df, "interval"] = count
274  index_df += 1
275 
276  interval_duration = left_interval - interval_time
277  acc_interval = df.loc[df["interval"] == count, "acc"]
278  mean_acc_interval = acc_interval.mean()
279 
280  interval_time_acc = interval_duration/mean_acc_interval
281  interval_time = left_interval
282 
283  if interval_time_acc < min_acc_scene_duration:
284  scale_factor = min_acc_scene_duration / interval_time_acc
285  df.loc[df["interval"] == count, "acc"] = acc_interval/scale_factor
286 
287  #Correct possible errors of less than min_video_duration, average set of accelerations
288  df = correct_acc_min_video_duration(df)
289 
290  return df

298 def correct_groups_acc_interval(df):

300  previous_acc = df.loc[0, 'acc']

394  acc_div = 1/acc_min
