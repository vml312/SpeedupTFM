92  for sub in subs:
93  if sub.text == "voice":
94  start_s = sub.start/1000
95  end_s = sub.end/1000
96 
97  current_acc = 0

99  i = 1
100  while i <= len(df) and df.loc[i, 'end-time-s']<= end_s:

103  n_voice_subs_1s += 1

115
