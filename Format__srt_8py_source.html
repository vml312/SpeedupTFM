1 """
2 Analysis of the subtitle file
3 
4 With the subtitle file resulting from the inaSpeechSegmenter analysis or with the subtitle file entered (depending on the option chosen), the subtitle file is checked for errors and
5 a new srt file is created (with the format of the speech or non-speech subtitles (voice or else)).
6 
7 The first phase is to fill the new file, if there are subtitles in a time period, the interval is created with the 'speech' tag. On the other hand, if there are no subtitles in the
8 interval, the 'no speech' tag is assigned. This is done in the file ''voice-else_subs.srt''.
9 If in the 'speech' to 'non-speech' transition, the 'non-speech' interval lasts less than one second, it is not separated, it remains as 'speech'.
10 
11 The second phase compresses this file, joining the consecutive speech and non-speech fragments, generating the file ''compr_subs.srt''.
12 
13 Finally, the total number of fragments to be generated is determined and this value is returned.
14 """

44 def fill_srt(file_name, n_segs_threshold):
45  subs = pysrt.open(file_name)
46  try:
47  file = open("kframes.txt", "r")
48  except FileNotFoundError:
49  print("File not found")
50  line = file.readline()
51  firstkframe = float(line.split(",")[0])
52 
53  last_hour = int(firstkframe / 3600)
54  last_minutes = int((float(firstkframe / 3600) - last_hour) * 60)
55  last_seconds = ((float(firstkframe / 3600) - last_hour) * 60 - last_minutes) * 60
56  last_end = [int(last_hour), int(last_minutes), float(last_seconds)] # h/min/s
57 
58  index = 1
59  with open("voice-else_subs.srt", 'w') as output:
60  for sub in subs:
61  content = str(sub)
62  content = content.splitlines()
63 
64  start_h = int(re.findall(r'(\d+\d+):', content[1])[0])
65  start_min = int(re.findall(r':(\d+\d+):', content[1])[0])
66  start_s = re.findall(r':(\d+\d+,\d+\d+\d+)', content[1])[0]
67  start_s = float(start_s.replace(',', '.'))
68 
69  end_h = int(re.findall(r'(\d+\d+):', content[1])[2])
70  end_min = int(re.findall(r':(\d+\d+):', content[1])[1])
71  end_s = re.findall(r':(\d+\d+,\d+\d+\d+)', content[1])[1]
72  end_s = float(end_s.replace(',', '.'))
73 
74  current_start = [start_h, start_min, start_s]
75  current_end = [end_h, end_min, end_s]
76 
77  if current_start == last_end:
78  output.write(str(index) + '\n' + content[1] + '\n' + "voice" + '\n\n')
79  index += 1
80  last_end = current_end
81  else:
82  # we have a gap between last end and current start => if the gap is longer than %n_segs_threshold% seconds => separate it
83  gap_duration = (start_h * 3600 + start_min * 60 + start_s) - (last_end[0] * 3600 + last_end[1] * 60 +
84  last_end[2])
85  if gap_duration > n_segs_threshold:
86  seconds1 = "{:06.3f}".format(last_end[2]).replace('.', ',')
87  seconds2 = "{:06.3f}".format(current_start[2]).replace('.', ',')
88  new_content = f'{last_end[0]:02}:{last_end[1]:02}:{seconds1} --> {current_start[0]:02}:{current_start[1]:02}:{seconds2}'
89  output.write(str(index) + '\n' + new_content + '\n' + "else" + '\n\n')
90  index += 1
91  # after writing the gap info, we'll also write the current object
92  output.write(str(index) + '\n' + content[1] + '\n' + "voice" + '\n\n')
93  index += 1
94  last_end = current_end
95 
96  else:
97  # special case for first subtitle
98  if current_start == [0, 0, 0]:
99  if current_end[0] <= last_end[0] and current_end[1] <= last_end[1] and current_end[2] <= \
100  last_end[2]:
101  print("end time < start time -> this output was eliminated and replaced")
102  else:
103  seconds1 = "{:06.3f}".format(last_end[2]).replace('.', ',')
104  seconds2 = "{:06.3f}".format(current_end[2]).replace('.', ',')
105  new_content = f'{last_end[0]:02}:{last_end[1]:02}:{seconds1} --> {current_end[0]:02}:{current_end[1]:02}:{seconds2}'
106  output.write(str(index) + '\n' + new_content + '\n' + "voice" + '\n\n')
107  index += 1
108  last_end = current_end
109 
110  # for the last fragment - final index
111  with open('kframes.txt', 'r') as f:
112  last_line = f.readlines()[-1]
113  lastkeyframe = float(last_line.split(",")[0])
114 
115  final_hour = int(lastkeyframe / 3600)
116  final_minutes = int((float(lastkeyframe / 3600) - final_hour) * 60)
117  final_seconds = ((float(lastkeyframe / 3600) - final_hour) * 60 - final_minutes) * 60
118  final_time = [int(final_hour), int(final_minutes), float(final_seconds)] # h/min/s
119  if last_end != final_time:
120  with open("voice-else_subs.srt", 'a') as output:
121  seconds1 = "{:06.3f}".format(last_end[2]).replace('.', ',')
122  seconds2 = "{:06.3f}".format(final_time[2]).replace('.', ',')
123  new_content = f'{last_end[0]:02}:{last_end[1]:02}:{seconds1} --> {final_time[0]:02}:{final_time[1]:02}:{seconds2}'
124  output.write(str(index) + '\n' + new_content + '\n' + "else" + '\n\n')
125  index += 1
